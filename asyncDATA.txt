async asyncData ({ app }) {
    const { data } = await app.$axios.post(process.env.POSTS_URL,
    JSON.stringify({
        filter: { published: true },
        sort: {_created:-1},
        populate: 1
      }),
    {
      headers: { 'Content-Type': 'application/json' }
    })
	const [gallery] = data.entries
    
    return { 
      posts: data.entries,
      gallery: gallery.gallery.length
    }    
  }
